 50     function Clone(baseObj, /** Object= */properties, /** PropertyDescriptor= */defaultDescriptor){
 51 
 52         if(this && this !== (global || window) ){
 53             // Called as constructor (with new operator or used call/apply):
 54             // We assume that the new object is already created,
 55             // and we need only to set its properties (if specified).
 56             if(arguments.length){
 57                 Array.prototype.unshift.call(arguments, this);
 58                 Function.call.apply(Clone.prototype.defineProperties, arguments);
 59             }
 60             return this;
 61 
 62         }else{
 63         // Called like a regular function:
 64             if(arguments.length == 0)
 65                 baseObj = Clone.prototype;
 66 
 67             //if(Clone.can(baseObj, 'clone').like(Clone) )
 68 
 69             return Clone.prototype.clone.call(baseObj, properties, defaultDescriptor);
 70         }

 74     /**
 75      * Translate object to property descriptors.
 76      * <p>For example, {a:{}, b:2} will be translated to something like {a: {value: {}}, b: {value: 2}}.
 77      * <p>Functions, properties prefixed with "_", and constructor will be automatically marked as non-enumerable.
 78      *
 79      * @param {Object} properties
 80      * @param {PropertyDescriptor=} defaultDescriptor The default property descriptor for properties.
 81      * @returns {{PropertyDescriptor}} Properties descriptors.
 82      *
 83      * @static
 84      */
 85     Clone.describe = function(properties, defaultDescriptor){
 86         var descriptors = {};
 87 
 88         if(!defaultDescriptor){
 89             /*
 90              * @typedef {Object}
 91              * @name PropertyDescriptor */
 92             defaultDescriptor = {
 93                 value: undefined,
 94                 configurable: true,
 95                 writable: true,
 96                 enumerable: true
 97             };
 98         }
 99 
100         for(var property in properties){
101             var value = properties[property];
102             var descriptor = descriptors[property] = Object.create(defaultDescriptor);
103             descriptor.value = value;
104 
105             // methods and private properties should be non-enumerable:
106             if(typeof(value)=='function' || property[0]=='_'){
107                 descriptor.enumerable = false;
108             }
109         }
110         if( descriptors.hasOwnProperty('constructor'))
111             descriptors.constructor.enumerable = false;
112 
113         return descriptors;
114     };

199     /**
200      * Use this method to check if method of one object stored in another object.
201      * @example
202      *     var myObj = {split: function(){} };
203      *     Clone.can(myObj, 'split').like(new Array);
204      *
205      * @returns {{valueOf: Function, like: function(Object), as: function(Object)}}
206      */
207     Clone.can = function(/** Object */obj1, /** string */method, /** number=false */not){
208         return {
209             valueOf: function(){
210                 // a ^ b == a ? !b : b
211                 return not ^ typeof( obj1[method] )=='function';
212             },
213 
214             like: function(obj2){
215                 var obj2Method = obj2[method];
216                 return this.valueOf()
217                     && not ^ (
218                         typeof(obj2Method)=='function' &&
219                         obj2Method.length == obj1[method].length
220                     );
221             },
222 
223             as: function(obj2){
224                 return this.valueOf()
225                     && not ^ obj1[method] === obj2[method];
226             }
227         }
228     };
229     /** @see Clone.can */
230     Clone.cant = function(/** Object */obj1, /** string */method){
231         return Clone.can(obj1, method, 1);

234     /**
235      * Mix two or more objects.
236      *
237      * @param root
238      *        The root object. Will be modified, if copyNesting does not set.
239      * @param objectToMix
240      *        Object(s) to mix.
241      * @param parentsLevel
242      *        Set this to Infinity to copy all parents properties up to Object.prototype.
243      * @param copyNesting
244      *        If not true, all properties of all objects will be directly attached to the one root object.
245      * @returns {Object}
246      */
247     Clone.mix = function(
248         /**       Object|FunctionType */root,
249         /** Object|FunctionType|Array */objectToMix,
250         /**                  number=0 */parentsLevel,
251         /**              boolean=true */copyNesting
252     ){
253         if( typeof(root)=='function' && Object.getOwnPropertyNames(root.prototype) ){
254             root = root.prototype;
255         }
256         if( copyNesting === undefined ){
257             copyNesting = true;
258         }
259 
260         if(objectToMix instanceof Array){
261             mixedObjects = objectToMix;
262             //objectToMix = mixedObjects.shift();
263         }else{
264             // get parents:
265             var obj = objectToMix;
266             var mixedObjects = [];
267             do  mixedObjects.push(obj);
268             while( parentsLevel-- && (obj = Object.getPrototypeOf(obj)) != Object.prototype);
269         }
270 
271         mixedObjects = mixedObjects.reverse();
272 
273         var updateObj = root;
274 
275         for(var objIdx in mixedObjects){
276             obj = mixedObjects[objIdx];
277 
278             if( typeof(obj)=='function' && Object.getOwnPropertyNames(obj.prototype) ){
279                 obj = obj.prototype;
280             }
281 
282             var ownPropertyNames = Object.getOwnPropertyNames(obj);
283 
284             if(copyNesting && ownPropertyNames.length){
285                 updateObj = Clone(updateObj);
286             }
287 
288             // copy all own properties from obj to updateObj:
289             for(var i=0; i < ownPropertyNames.length; i++){
290                 var name = ownPropertyNames[i];
291                 var descriptor = Object.getOwnPropertyDescriptor(obj, name);
292                 Object.defineProperty(updateObj, name, descriptor);
293             }
294         }
295 
296         return updateObj;
297     };

364          * Apply method of super object to this object.
365          * @returns {*}

367         applySuper: function(/** string=constructor|Array */ methodName, /** Array= */args){
368             if(typeof(methodName) != 'string'){
369                 if( arguments[0] instanceof Array){
370                     args = arguments[0];
371                 }
372                 methodName = 'constructor';
373             }//</arguments>
374 
375             /* if not */('__super__' in this) || this.defineProperty(
376                 '__super__', {value: Object.getPrototypeOf(Object.getPrototypeOf(this)), writable:!0,configurable:!0}
377             );
378 
379             // save super
380             var savedSuper = this.__super__;
381             // set super to next by prototype chain, in case if method also call applySuper
382             this.__super__ = Object.getPrototypeOf(savedSuper);
383             // apply method
384             var returned   = savedSuper[methodName].apply(this, args);
385             // restore super
386             this.__super__ = savedSuper;
387 
388             return returned;
389         },

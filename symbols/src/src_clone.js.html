  3  * @title   clone.js - The true prototype-based OOP framework.
  4  * @version 0.6.1 alpha
  5  * @author  Alex Shvets
  6  * @see     <a href="http://quadroid.github.com/clonejs/" >Documentation</a>
  7  * @see     <a href="http://github.com/quadroid/clonejs/" >GitHub</a>
  8  *
  9  * @class
 10  *     This is the framework that implements the true prototype-based OOP paradigm in JS.<br>
 11  *     It's based on the new ECMA Script 5 features like Object.create and property descriptors.
 12  *
 13  *     <p><a href="http://github.com/quadroid/clonejs">Project on GitHub</a>
 14  *
 15  *     <p><b>Naming conventions</b>
 16  *     <p>Var names, <b>prefixed by "$"</b>, contain object, used as prototype for other objects. For example:
 17  *     <code>
 18  *     var $array = Array.prototype, $myType = {},
 19  *         myTypeInstance = Object.create($myType);// $object.apply('create', $myType);
 20  *     </code>
 21  *     <p>Properties, <b>prefixed by "_"</b>, are private.
 22  *     <p>
 23  * @description
 24  *     With this library you should forget about classes. Use prototypes instead.
 25  *
 26  * @example
 27  *     var $myType = $object.clone({
 28  *         '(final)  notConfigurableAndNotWritable': true,
 29  *         '(writable final)   notConfigurableOnly': null,
 30  *         '(hidden final get) notEnumerableGetter': function(){},
 31  *         '(hidden)                 notEnumerable': true,
 32  *         '(const)                       constant': 'not writable',
 33  *                                  publicProperty : 1,
 34  *                                           _item : null,// private property (not enumerable)
 35  *                                     '(get) item': function() { return this._item },
 36  *                                     '(set) item': function(v){ this._item = v },
 37  *                      '(get) publicPropertyAlias': 'publicProperty',// automatically create getter for publicProperty
 38  *                                     constructor : function MyType(){
 39  *                                                       var obj = this.applySuper(arguments);
 40  *                                                       // do something with obj...
 41  *                                                       return obj;
 42  *                                                   }
 43  *     });
 44  *     var myTypeInstance = $myType.create({publicProperty: 2});
 45  *     assert( $myType.isPrototypeOf(myTypeInstance) );
 46  *     assert( $myType.publicPropertyAlias === $myType.publicProperty );
 47  *
 48  *     var $myArray1 = $object.clone.call(Array.prototype, {customMethod: function(){}});
 49  *     var $myArray2 = $object.copy(Array).setProperties({customMethod: function(){}});
 50  *
 51  *     var myObj = {a:1, b:2, c:3};
 52  *     var cloneOfMyObj = $object.apply('clone', myObj);
 53  *     cloneOfMyObj.a = 11; // myObj.a still == 1
 54  */

 58      * Create a clone of object. See <a href="http://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/create">Object.create</a> for details.
 59      * @see <a href="http://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/create">Object.create</a>
 60      * @returns {$object}
 61      * @memberOf $object#
 62      */
 63     clone: function(/** Object= */properties, /** PropertyDescriptor= */defaultDescriptor){
 64         if(arguments.length){
 65             var descriptors = $object.describe.apply(null, arguments);
 66 
 67             if( descriptors.hasOwnProperty('constructor') ){
 68                 var constructor = descriptors.constructor.value;
 69                 if(typeof constructor == 'string'){
 70                     descriptors.constructor.value = function CustomType(){
 71                         return this.applySuper(arguments);
 72                     }
 73                 }
 74                 descriptors.constructor.value.prototype = this;
 75             }
 76         }
 77         return Object.create(this, descriptors);
 78     },

117     /**
118      * Translate object to property descriptors.
119      * <p>For example, {a:{}, b:2} will be translated to something like {a: {value: {}}, b: {value: 2}}.
120      * <p>Functions (except getters) and properties prefixed by "_" will be automatically marked as non-enumerable.
121      * <p>You can prefix your property names by (get|set|const|final|hidden|writable).
122      *     <li>(get) - define getter, if string passed, the getter will be auto generated.
123      *     <li>(set) - define setter, if string passed, the setter will be auto generated.
124      *     <li>(const) - make property unwritable.
125      *     <li>(final) - make property unwritable, and prevent it deleting and descriptor modifications.
126      *     <li>(hidden) - make property non-enumerable.
127      *     <li>(writable) - make property writable (use with final).
128      *
129      * @param {Object} properties
130      * @param {PropertyDescriptor=} defaultDescriptor The default property descriptor.
131      * @returns {{PropertyDescriptor}} Property descriptors.
132      *
133      * @static
134      * @memberOf $object
135      */
136     describe: function(properties, defaultDescriptor){
137         var descriptors = {};
138 
139         var $defaultDescriptor = defaultDescriptor ? defaultDescriptor : {
140             configurable: true,
141             enumerable: true,
142             writable: true
143         };
144 
145         var hidingAllowed = !(defaultDescriptor && defaultDescriptor.enumerable);
146 
147         for(var name in properties){
148             var value = properties[name];
149             var descriptor = Object.create($defaultDescriptor);
150 
151             if( name[0]=='(' ){
152                 //TODO: fix regexp to not mach the '(getset) property'
153                 var matches = name.match(/^\((((get|set|const|hidden|final|writable) *)+)\) +(.+)$/);
154                 if( matches ){
155                     var prefixes = matches[1].split(' ').sort();
156                     name = matches[4];
157 
158                     if(descriptors[name]) descriptor = descriptors[name];
159 
160                     for(var i in prefixes) switch(prefixes[i]){
161                         case    'const': descriptor.writable     = false; break;
162                         case    'final': descriptor.configurable = false; descriptor.writable = false; break;
163                         case      'get': descriptor.get          = value; break;
164                         case   'hidden': descriptor.enumerable   = false; break;
165                         case      'set': descriptor.set          = value; break;
166                         case 'writable': descriptor.writable     = true;  break;
167                     }
168                 }
169             }
170 
171             if(descriptor.get || descriptor.set){
172                 if(typeof value == 'string'){
173                     var hiddenPropertyName = value;
174                     if(typeof descriptor.get == 'string'){
175                         descriptor.get = function getter(){
176                             return this[hiddenPropertyName];
177                         }
178                     }else{
179                         descriptor.set = function setter(newValue){
180                             this[hiddenPropertyName] = newValue;
181                         }
182                     }
183                 }
184                 descriptor.value = undefined;
185                 if(descriptor.get) value = undefined;// do not allow to hide getter by default
186             }else{
187                 descriptor.value = value;
188             }
189 
190             // hide methods and private properties:
191             if(hidingAllowed && typeof(value)=='function' || name[0]=='_'){
192                 descriptor.enumerable = false;
193             }
194     //        // constants:
195     //        if(name.toUpperCase() == name){
196     //            descriptor.writable = false;
197     //        }
198             descriptors[name] = descriptor;
199         }
200     //    if( descriptors.hasOwnProperty('constructor')){
201     //        descriptors.constructor.enumerable = false;
202     //    }
203         return descriptors;
204     },

206     /**
207      * Apply method of super object (prototype) to this object.
208      * @returns {*}
209      * @see $object#__super__
210      * @see $object#callSuper
211      * @protected
212      * @memberOf $object#
213      */
214     applySuper: function(/** string='constructor'|Array */ methodName, /** Array= */args){
215         if(typeof(methodName) != 'string'){
216             if( arguments[0] instanceof Array){
217                 args = arguments[0];
218             }
219             methodName = 'constructor';
220         }//</arguments>
221 
222         /* if not */('__super__' in this) || this.defineProperty(
223             '__super__', {value: Object.getPrototypeOf(Object.getPrototypeOf(this)), writable:!0,configurable:!0}
224             /**
225              * Link to the object prototype.
226              * Dynamically changed to next by prototype chain, while applySuper method executing.
227              * System property. <b>Use it only for debug purposes</b>.
228              * @name  __super__
229              * @type  {?Object}
230              * @see $object#applySuper
231              * @private
232              * @memberOf $object#
233              */
234         );
235 
236         // save super
237         var savedSuper = this.__super__;
238         // set super to next by prototype chain, in case if method also call applySuper
239         this.__super__ = Object.getPrototypeOf(savedSuper);
240         // apply method
241         var returned   = savedSuper[methodName].apply(this, args);
242         // restore super
243         this.__super__ = savedSuper;
244 
245         return returned;
246     },

330     /**
331      * Return new object, that have a copies of all own properties of this object. <br>
332      * Arguments can be passed in any order (recognized by type).
333      * @example
334      *    var $collection = $object.clone(),
335      *        $users = $collection.clone();
336      *
337      *    // $users full prototype chain:
338      *    $users -> $collection -> $object -> Object.prototype -> null
339      *
340      *    // see prototype chains produced by copy:
341      *
342      *    $users.copy():
343      *        ~$users -> $object
344      *
345      *    $users.copy(rootPrototype:Array):
346      *        ~$users -> Array.prototype
347      *
348      *    $users.copy(rootPrototype:$parent, parentsLevel:Infinity):
349      *        ~$users -> ~$collection -> ~$object -> $parent
350      *
351      *    $users.copy(rootPrototype:$parent, parentsLevel:Infinity, mixParents:true):
352      *        ~($users + $collection + $object) -> $parent
353      *
354      *    // where ~$users:
355      *    New plain object, that have a copy of every own $user property.
356      *
357      * @param deepMethod
358      *        How to process inner objects. Can be:         <br>
359      *        "deepCopy"  - see {@link $object#deepCopy}    <br>
360      *        "deepClone" - see {@link $object#deepClone}   <br>
361      *        null - do nothing (default).
362      *
363      * @param rootPrototype
364      *        The root prototype for created object prototype chain. If it is Constructor, Constructor.prototype will be used instead. <br>
365      *        By default - $object;
366      *
367      * @param parentsLevel
368      *        How many parents should be included. By default - zero. <br>
369      *        Set this to Infinity if you want to copy all object parents properties up to $object.
370      *
371      * @param mixParents
372      *        Should be false if objects have methods, that call {@link $object#applySuper} <br>
373      *        If true, all own properties of all objects will be directly attached to the one returned object. <br>
374      *        False by default.
375      *
376      * @returns {Object}
377      * @memberOf $object#
378      */
379     copy: function(
380         /** ('deepClone'|'deepCopy')= */deepMethod,
381         /**     (Object|Constructor)= */rootPrototype,
382         /**                  number=0 */parentsLevel,
383         /**             boolean=false */mixParents
384         //**                Array=all */propertiesList
385         ){
386         for(var i=0, value=arguments[i]; i < arguments.length; value=arguments[++i]) switch(typeof value){
387             case 'string':   deepMethod    = value; break;
388             case 'function': rootPrototype = value.prototype; break;
389             case 'object':   rootPrototype = value; break;
390             case 'number':   parentsLevel  = value; break;
391             case 'boolean':  mixParents    = value; break;
392         }
393 
394         if( deepMethod != 'deepCopy' || deepMethod != 'deepClone'){
395             deepMethod = null;
396         }
397 
398         if(typeof rootPrototype == 'undefined'){
399             rootPrototype = $object;
400         }
401 
402         var sourceObj = this;
403         if( typeof(sourceObj)=='function' && Object.getOwnPropertyNames(sourceObj.prototype) ){
404             sourceObj = sourceObj.prototype;
405         }
406         var newObj    = Object.create(rootPrototype);
407         var updateObj = newObj;
408 
409         var sourceObjects = [];
410         do{
411             sourceObjects.push(sourceObj);
412             sourceObj = Object.getPrototypeOf(sourceObj);
413         }while(parentsLevel-- && sourceObj != Object.prototype);
414 
415         sourceObjects = sourceObjects.reverse();
416 
417         for(var i=0; i < sourceObjects.length; i++){
418             sourceObj = sourceObjects[i];
419             var ownPropertyNames = Object.getOwnPropertyNames(sourceObj);
420             if (!mixParents && i && ownPropertyNames.length){
421                 updateObj = $object.apply('clone', updateObj);
422             }
423 
424             // copy all own properties:
425             for(var p=0; p < ownPropertyNames.length; p++){
426                 var name = ownPropertyNames[p];
427                 var descriptor = Object.getOwnPropertyDescriptor(sourceObj, name);
428 
429                 if( deepMethod && typeof sourceObj[name] == 'object'){
430                     descriptor.value = $object[deepMethod].call(sourceObj[name]);
431                 }
432 
433                 Object.defineProperty(updateObj, name, descriptor);
434             }
435         }
436 
437         return updateObj;
438     },

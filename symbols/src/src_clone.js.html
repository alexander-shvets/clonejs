1 'use strict';
  2 /**
  3  * @title   clone.js - the true prototype-based JavaScript micro-framework.
  4  * @version v0.7.2-05-beta
  5  * @author  Alex Shvets
  6  *
  7  * @class
  8  * This is the framework that implements the true [prototype-based OOP⠙][1] paradigm in JS.  
  9  * It's based on the ECMA Script 5 features like [Object.create⠙][2] and [property descriptors⠙][3].
 10  *    [1]: http://en.wikipedia.org/wiki/Prototype-based_programming
 11  *    [2]: https://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/create
 12  *    [3]: https://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/defineProperty
 13  *
 14  * <p>[View on GitHub](http://github.com/quadroid/clonejs#readme)
 15  *
 16  * ### Naming conventions
 17  *
 18  * Var names, **prefixed by "$"**, contain object, used as prototype for other objects. 
 19  * For example:
 20  * <code>
 21  * var $array = Array.prototype, $myType = {},
 22  *     myTypeInstance = Object.create($myType);// $object.apply($myType, 'create');
 23  * </code>
 24  *
 25  * Properties, **prefixed by "_"**, are private.
 26  *
 27  * [![githalytics.com alpha](https://cruel-carlota.pagodabox.com/3110be9614da5cb337ebd483c187010f "githalytics.com")](http://githalytics.com/quadroid/clonejs)
 28  *
 29  * @description
 30  * The main difference with other class-producing tools like `Ext.define`, `dojo.declare`, `Backbone.Model.extend`
 31  * is that `$object.clone` will return an object (prototype with defined constructor-function) instead of function
 32  * (with defined prototype-object). So, you don't need for instantiation, you can just start using the cloned object right now.
 33  * But, if you need more than one instance, you can create it by `$yourProto.create`.
 34  *
 35  * @example
 36  *
 37  * var myObj = {a:1, b:2, c:3};
 38  * var cloneOfMyObj = $object.apply(myObj, 'clone');
 39  * cloneOfMyObj.a = 11; // myObj.a still == 1
 40  * myObj.b = 22; // cloneOfMyObj.b will be also changed to 22
 41  *
 42  * var $myType = $object.clone({
 43  *     '(final)          property1': "not configurable and not writable",
 44  *     '(writable final) property2': "not configurable only",
 45  *     '(hidden)         property3': "not enumerable",
 46  *     '(const)           constant': "not writable",
 47  *                       property4 : "simple property",
 48  *     '(get)       property3alias': 'property3',// automatically create getter
 49  *                           _item : "private property (not enumerable)",
 50  *                    '(get)  item': function() { return this._item },
 51  *                    '(set)  item': function(v){ this._item = v    },
 52  *                     constructor : function MyType(){
 53  *                                       this.applySuper(arguments);
 54  *                                       // do something...
 55  *                                   }
 56  * });
 57  * assert( $myType.property3alias === $myType.property3 );
 58  * 
 59  * var myTypeInstance = $myType.create({property4: "initialize simple property"});
 60  * assert( $myType.isPrototypeOf(myTypeInstance) );
 61  *
 62  * var $myArray1 = $object.clone.call(Array.prototype, {customMethod: function(){}});
 63  * var $myArray2 = $object.copy(Array).setProperties({customMethod: function(){}});
 64  *
 65  */

 88     /**
 89      * Use this method to create an instances of prototype objects.  
 90      * Behaves like a [clone](#clone) method. But also apply constructor, and, if [default constructor][1] called,
 91      * the created instance will be [sealed⠙][2] to avoid creation of [hidden classes⠙][3].  
 92      * All arguments, passed to `create()`, will be forwarded to [constructor][1].
 93      * [1]: #constructor
 94      * [2]: http://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/seal
 95      * [3]: https://developers.google.com/v8/design#prop_access
 96      * @see $object#clone
 97      * @see $object#constructor
 98      * @see $object.describe

100      * @returns {$object}

102      * @example
103      *     var $myType = $object.clone({
104      *         constructor: 'MyType',
105      *     });
106      *     var myTypeInstance = $myType.create();
107      *     assert( myTypeInstance.constructor === $myType.constructor );
108      *     assert( $myType.isPrototypeOf(myTypeInstance) );

110      * @memberOf $object#
111      */

113         var obj = this.clone();
114         return obj.constructor.apply(obj, arguments) || obj;

134     /**
135      * Translate object to property descriptors.
136      * For example, `{a: {b:1}, c:2}` will be translated to `{a: {value: {b:1}}, b: {value: 2}}`.  
137      *  
138      * Functions (except getters) and properties prefixed by "_" will be automatically marked as non-enumerable. 
139      *  
140      * Uppercase properties will be not writable. 
141      *  
142      * You can prefix your property names by `(get|set|const|final|hidden|writable)`:
143      * <dl class="detailList params dl">
144      * <dt>(get)     <dt><dd> define getter, if string passed, the getter will be auto generated.
145      * <dt>(set)     <dt><dd> define setter, if string passed, the setter will be auto generated.
146      * <dt>(const)   <dt><dd> make property unwritable.
147      * <dt>(final)   <dt><dd> make property unwritable, and prevent it deleting and descriptor modifications.
148      * <dt>(hidden)  <dt><dd> make property non-enumerable.
149      * <dt>(writable)<dt><dd> make property writable (use with final). 
150      * </dl>
151      * @param properties
152      * @param defaultDescriptor The default property descriptor.
153      * @returns {{PropertyDescriptor}} Property descriptors.
154      * @this {Object} Prototype only.
155      * @static
156      * @memberOf $object
157      */
158     describe: function(/** Object */properties, /** PropertyDescriptor= */defaultDescriptor){
159         
160         var descriptors = {};
161         
162         /// Default properties descriptor:
163 
164         var $defaultDescriptor = defaultDescriptor ? defaultDescriptor : {
165             configurable: true,
166             enumerable: true,
167             writable: true
168         };
169         
170         /// Iterate properties:
171 
172         var hidingAllowed = !(defaultDescriptor && defaultDescriptor.enumerable);
173 
174         for(var name in properties){
175             
176             var value = properties[name];
177             var descriptor = Object.create($defaultDescriptor);
178 
179             /// apply property modifiers:
180             
181             if( name[0]=='(' ){
182                 //TODO: fix regexp to not mach the '(getset) property'
183                 var matches = name.match(/^\((((get|set|const|hidden|final|writable) *)+)\) +(.+)$/);
184                 if( matches ){
185                     var prefixes = matches[1].split(' ').sort();
186                     name = matches[4];
187 
188                     if(descriptors[name]) descriptor = descriptors[name];
189 
190                     for(var i in prefixes) switch(prefixes[i]){
191                         case    'const': descriptor.writable     = false; break;
192                         case    'final': descriptor.configurable = false; descriptor.writable = false; break;
193                         case      'get': descriptor.get          = value; break;
194                         case   'hidden': descriptor.enumerable   = false; break;
195                         case      'set': descriptor.set          = value; break;
196                         case 'writable': descriptor.writable     = true;  break;
197                     }
198                 }
199             }
200             
201             /// define getters/setters:
202             
203             if(descriptor.get || descriptor.set){
204                 if(typeof value == 'string'){
205                     var hiddenPropertyName = value;
206                     if(typeof descriptor.get == 'string'){
207                         descriptor.get = function getter(){
208                             return this[hiddenPropertyName];
209                         }
210                     }else{
211                         descriptor.set = function setter(newValue){
212                             this[hiddenPropertyName] = newValue;
213                         }
214                     }
215                 }
216                 descriptor.value = undefined;
217                 if(descriptor.get) value = undefined;// do not allow to hide getter as method
218             }else{
219                 descriptor.value = value;
220             }
221 
222             /// hide methods and private properties:
223             
224             if(hidingAllowed && typeof(value)=='function' || name[0]=='_'){
225                 descriptor.enumerable = false;
226             }
227 
228             /// constants:
229             
230             if(name.toUpperCase() == name){
231                 descriptor.writable = false;
232             }
233             
234             ///
235             
236             descriptors[name] = descriptor;
237         }
238         
239         /// constructor:
240         
241         if( descriptors.hasOwnProperty('constructor') ){
242             var $prototype = this;
243             var constructor = descriptors.constructor.value;
244             if(typeof constructor == 'string'){
245                 var typeName = constructor;
246                 // isn't slow, see http://jsperf.com/eval-vs-new-function-vs-function
247                 constructor = Function(
248                     'return function '+typeName+'(){return this.applySuper(arguments)}'
249                 )();
250                 constructor.typeName = typeName;
251                 descriptors.constructor.value = constructor;
252 
253             }else if(!constructor.typeName){
254                 constructor.typeName = constructor.name || 'CloneOf'+$prototype.constructor.typeName;//+n
255             }
256 
257             constructor.prototype = $prototype;
258             descriptors.constructor.enumerable = false;
259         }
260 
261         ///
262         
263         return descriptors;
264     },

266     /**
267      * Apply method of super object (prototype) to this object.
268      * @returns {*}
269      * @see $object#__super__
270      * @see $object#callSuper
271      * @this {Object} Instance only.
272      * @protected
273      * @memberOf $object#
274      */
275     applySuper: function(/** Array|string='constructor' */ methodName, /** Array= */args){
276         if(typeof(methodName) != 'string'){
277             args = arguments[0];
278             methodName = 'constructor';
279         }//</arguments>
280 
281         /* if not */('__super__' in this) || this.defineProperty(
282             '__super__', {value: this.getSuper(), writable:true, configurable:true}
283             /**
284              * Link to the instance prototype. 
285              * Dynamically changed to next by prototype chain, while `{@link #applySuper}` method executing. 
286              * System property. **Use it only for debug purposes**.
287              * @name  __super__
288              * @type  {?Object}
289              * @see $object#applySuper
290              * @private
291              * @memberOf $object#
292              */
293         );
294 
295         // save super
296         var savedSuper = this.__super__;
297         // set super to next by prototype chain, in case if method also call applySuper
298         this.__super__ = Object.getPrototypeOf(savedSuper);
299         // apply method
300         var returned   = savedSuper[methodName].apply(this, args);
301         // restore super
302         this.__super__ = savedSuper;
303 
304         return returned;
305     },

380     /**
381      * Return new object, that have a copies of all own properties of this object.  
382      * Arguments can be passed in any order (recognized by type).
383      * @example
384      * var $collection = $object.clone(),
385      *     $users = $collection.clone();
386      *
387      * // $users full prototype chain:
388      * $users -> $collection -> $object -> Object.prototype -> null
389      *
390      * // see prototype chains produced by copy:
391      *
392      * $users.copy():
393      *     ~$users -> $object
394      *
395      * $users.copy(rootPrototype:Array):
396      *     ~$users -> Array.prototype
397      *
398      * $users.copy(rootPrototype:$parent, parentsLevel:Infinity):
399      *     ~$users -> ~$collection -> ~$object -> $parent
400      *
401      * $users.copy(rootPrototype:$parent, parentsLevel:Infinity, mixParents:true):
402      *     ~($users + $collection + $object) -> $parent
403      *
404      * // where ~$users:
405      * New plain object, that have a copy of every own $user property.
406      *
407      * @this {Object} Instance or prototype.
408      * 
409      * @param deepMethod
410      *        How to process inner objects. Can be:        
411      *        "deepCopy"  - see `{@link #deepCopy}`   
412      *        "deepClone" - see `{@link #deepClone}`  
413      *        "" - do nothing (default).
414      *
415      * @param rootPrototype
416      *        The root prototype for created object prototype chain. If it is Constructor, Constructor.prototype will be used instead.  
417      *        By default - $object;
418      *
419      * @param parentsLevel
420      *        How many parents should be included. By default - zero.  
421      *        Set this to Infinity if you want to copy all object parents properties up to $object.
422      *
423      * @param mixParents
424      *        Should be false if objects have methods, that call `{@link #applySuper}`.
425      *        If true, all own properties of all objects will be directly attached to the one returned object.  
426      *        False by default.
427      *
428      * @returns {Object}
429      * @memberOf $object#
430      */
431     copy: function(
432         /** ('deepClone'|'deepCopy')=""      */deepMethod,
433         /**     (Object|Constructor)=$object */rootPrototype,
434         /**                   number=0       */parentsLevel,
435         /**                  boolean=false   */mixParents
436         //**                Array=all */propertiesList
437     ){
438         for(var i=0, value=arguments[i]; i < arguments.length; value=arguments[++i]) switch(typeof value){
439             case 'string':   deepMethod    = value; break;
440             case 'function': rootPrototype = value.prototype; break;
441             case 'object':   rootPrototype = value; break;
442             case 'number':   parentsLevel  = value; break;
443             case 'boolean':  mixParents    = value; break;
444         }
445 
446         if( deepMethod != 'deepCopy' || deepMethod != 'deepClone'){
447             deepMethod = null;
448         }
449 
450         if(typeof rootPrototype == 'undefined'){
451             rootPrototype = $object;
452         }
453 
454         var sourceObj = this;
455         if( typeof(sourceObj)=='function' && Object.getOwnPropertyNames(sourceObj.prototype) ){
456             sourceObj = sourceObj.prototype;
457         }
458         var newObj    = Object.create(rootPrototype);
459         var updateObj = newObj;
460 
461         var sourceObjects = [];
462         do{
463             sourceObjects.push(sourceObj);
464             sourceObj = Object.getPrototypeOf(sourceObj);
465         }while(parentsLevel-- && sourceObj != Object.prototype);
466 
467         sourceObjects = sourceObjects.reverse();
468 
469         for(var i=0; i < sourceObjects.length; i++){
470             sourceObj = sourceObjects[i];
471             var ownPropertyNames = Object.getOwnPropertyNames(sourceObj);
472             if (!mixParents && i && ownPropertyNames.length){
473                 updateObj = $object.apply(updateObj, 'clone');
474             }
475 
476             // copy all own properties:
477             for(var p=0; p < ownPropertyNames.length; p++){
478                 var name = ownPropertyNames[p];
479                 var descriptor = Object.getOwnPropertyDescriptor(sourceObj, name);
480 
481                 if( deepMethod && typeof sourceObj[name] == 'object' && sourceObj[name] !== null){
482                     descriptor.value = $object[deepMethod].call(sourceObj[name]);
483                 }
484 
485                 Object.defineProperty(updateObj, name, descriptor);
486             }
487         }
488 
489         return updateObj;
490     },

555     /**
556      * Copy properties from given object. 
557      * If allProperties does not specified, only own properties will be copied.
558      * @this {Object} Instance or prototype.
559      * @param obj
560      * @param allProperties Array of property names, or `true`. 
561      *        If true, all (own and parents) properties will be copied. 
562      * @returns {$object} this
563      * @see $object#paste
564      * @memberOf $object#
565      */
566     concat: function(/** Object */obj, /** (Array|boolean)= */allProperties){
567         if(allProperties == undefined) allProperties = obj.getOwnPropertyNames();
568         else if(typeof allProperties != 'object' && allProperties == true){
569             allProperties = obj.getOwnPropertyNames();
570             var proto = obj;
571             while( (proto = Object.getPrototypeOf(proto)) !== Object.prototype ){
572                 allProperties = allProperties.concat( proto.getOwnPropertyNames() );
573             }
574         }
575 
576         proto = obj;
577         do for(var i=0; i < allProperties.length; i++){
578             var name = allProperties[i];
579             var descriptor = Object.getOwnPropertyDescriptor(proto, name);
580             if( descriptor ){
581                 Object.defineProperty(this, name, descriptor);
582                 delete allProperties[i];
583             }
584         }
585         while(allProperties.length && (proto = Object.getPrototypeOf(proto)) !== Object.prototype);
586 
587         return this;
588     },

614     /**
615      * Use this to check if method of one object is the same as in the another object.
616      * @example
617      * var myObj1 = $object.clone({join: function(sep){} });
618      * var myObj2 = $object.clone({join: function(){}    });
619      * assert( myObj1.can('split').like( Array.prototype ) === true  );
620      * assert( myObj2.can('split').like( Array.prototype ) === false );
621      * assert( myObj1.can('split').as(   Array.prototype ) === false );
622      * assert( $object.can.call(new Array, 'split').as(new Array) === true  );
623      *
624      * @this {Object} Instance or prototype.
625      * @returns {{like: function(Object):boolean, as: function(Object):boolean}}
626      * @memberOf $object#
627      */
628     can: function(/** string */method, /** number=false */not){
629         var obj1 = this;
630         return {
631             /** @ignore *///jsdoc3
632             valueOf: function(){
633                 // a ^ b == a ? !b : b
634                 return not ^ typeof( obj1[method] )=='function';
635             },
636 
637             like: function(obj2){
638                 var obj2Method = obj2[method];
639                 return this.valueOf()
640                     && not ^ (
641                     typeof(obj2Method)=='function' &&
642                         obj2Method.length == obj1[method].length
643                     );
644             },
645 
646             as: function(obj2){
647                 return this.valueOf()
648                     && not ^ obj1[method] === obj2[method];
649             }
650         }

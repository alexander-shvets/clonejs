278         var proto = obj.__proto__ || obj;
279         if(descriptor.get || descriptor.set){
280             var PropertyName  = propertyName[0].toUpperCase() + propertyName.substring(1);
281             obj[propertyName] = function initAccessor_shim(value){
282                 if(value){
283                     obj["set" + PropertyName](value);
284                 }else{
285                     return obj["get" + PropertyName](value);
286                 }

288             if (descriptor.get){
289                 proto["get" + PropertyName] = descriptor.get;
290             }if(descriptor.set){

293         }else{
294             if( descriptor.enumerable ){

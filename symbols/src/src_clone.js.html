  5  * @title   clone.js - the true prototype-based JavaScript micro-framework.
  6  * @version v0.7.4-beta
  7  * @author  Alex Shvets
  8  *
  9  * @class
 10  * This is the framework that implements the true [prototype-based OOP⠙][1] paradigm in JS.  
 11  * It's based on the ECMA Script 5 features like [Object.create⠙][2] and [property descriptors⠙][3].
 12  *    [1]: http://en.wikipedia.org/wiki/Prototype-based_programming
 13  *    [2]: https://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/create
 14  *    [3]: https://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/defineProperty
 15  *
 16  * <p>[View on GitHub](http://github.com/quadroid/clonejs#readme)
 17  *
 18  * ### Naming conventions
 19  *
 20  * Var names, **prefixed by "$"**, contain object, used as prototype for other objects. 
 21  * For example:
 22  * <code>
 23  * var $array = Array.prototype, $myType = {},
 24  *     myTypeInstance = Object.create($myType);// $object.apply($myType, 'create');
 25  * </code>
 26  *
 27  * Properties, **prefixed by "_"**, are private.
 28  *
 29  * [![githalytics.com alpha](https://cruel-carlota.pagodabox.com/3110be9614da5cb337ebd483c187010f "githalytics.com")](http://githalytics.com/quadroid/clonejs)
 30  *
 31  * @description
 32  * The main difference with other class-producing tools like `Ext.define`, `dojo.declare`, `Backbone.Model.extend`
 33  * is that `$object.clone` will return an object (prototype with defined constructor-function) instead of function
 34  * (with defined prototype-object). So, you don't need for instantiation, you can just start using the cloned object right now.
 35  * But, if you need more than one instance, you can create it by `$yourProto.create`.
 36  *
 37  * @example
 38  *
 39  * var myObj = {a:1, b:2, c:3};
 40  * var cloneOfMyObj = $object.apply(myObj, 'clone');
 41  * cloneOfMyObj.a = 11; // myObj.a still == 1
 42  * myObj.b = 22; // cloneOfMyObj.b will be also changed to 22
 43  *
 44  * var $myType = $object.clone({
 45  *     '(final)          property1': "not configurable and not writable",
 46  *     '(writable final) property2': "not configurable only",
 47  *     '(hidden)         property3': "not enumerable",
 48  *     '(const)           constant': "not writable",
 49  *                       property4 : "simple property",
 50  *     '(get)       property3alias': 'property3',// automatically create getter
 51  *                           _item : "private property (not enumerable)",
 52  *                    '(get)  item': function() { return this._item },
 53  *                    '(set)  item': function(v){ this._item = v    },
 54  *                     constructor : function MyType(){
 55  *                                       this.applySuper(arguments);
 56  *                                       // do something...
 57  *                                   }
 58  * });
 59  * assert( $myType.property3alias === $myType.property3 );
 60  * 
 61  * var myTypeInstance = $myType.create({property4: "initialize simple property"});
 62  * assert( $myType.isPrototypeOf(myTypeInstance) );
 63  *
 64  * var $myArray1 = $object.clone.call(Array.prototype, {customMethod: function(){}});
 65  * var $myArray2 = $object.copy(Array).setProperties({customMethod: function(){}});
 66  *

135     /**
136      * Translate object to property descriptors.
137      * For example, `{a: {b:1}, c:2}` will be translated to `{a: {value: {b:1}}, b: {value: 2}}`.  
138      *  
139      * Functions (except getters) and properties prefixed by "_" will be automatically marked as non-enumerable. 
140      *  
141      * Uppercase properties will be not writable. 
142      *  
143      * You can prefix your property names by `(get|set|const|final|hidden|writable)`:
144      * <dl class="detailList params dl">
145      * <dt>(get)     <dt><dd> define getter, if string passed, the getter will be auto generated.
146      * <dt>(set)     <dt><dd> define setter, if string passed, the setter will be auto generated.
147      * <dt>(get once)<dt><dd> on first property call, getter fn will be called, and property will be replaced by returned value. 
148      * <dt>(const)   <dt><dd> make property unwritable.
149      * <dt>(final)   <dt><dd> make property unwritable, and prevent it deleting and descriptor modifications.
150      * <dt>(hidden)  <dt><dd> make property non-enumerable.
151      * <dt>(writable)<dt><dd> make property writable (use with final). 
152      * </dl>
153      * @param properties
154      * @param defaultDescriptor The default property descriptor.
155      * @returns {{PropertyDescriptor}} Property descriptors.
156      * @this {Object} Prototype only.
157      * @static
158      * @memberOf $object
159      */
160     describe: function(/** Object */properties, /** PropertyDescriptor= */defaultDescriptor){
161         
162         var descriptors = {};
163         
164         /// Default properties descriptor:
165 
166         var $defaultDescriptor = defaultDescriptor ? defaultDescriptor : {
167             configurable: true,
168             enumerable: true,
169             writable: true
170         };
171         
172         /// Iterate properties:
173 
174         var hidingAllowed = !(defaultDescriptor && defaultDescriptor.enumerable);
175 
176         for(var name in properties){
177             
178             var value = properties[name];
179             var descriptor = Object.create($defaultDescriptor);
180             var accessor = null;
181 
182             /// apply property modifiers:
183             
184             if( name[0]=='(' ){
185                 //TODO: fix regexp to not mach the '(getset) property'
186                 var matches = name.match(/^\((((get|set|once|const|hidden|final|writable) *)+)\) +(.+)$/);
187                 if( matches ){
188                     var modifiers = matches[1].split(' ').sort();
189                     name = matches[4];
190 
191                     if( descriptors[name] ){ 
192                         descriptor = descriptors[name];
193                     
194                     }else if( modifiers.indexOf('get') >= 0 || modifiers.indexOf('set') >= 0 ){
195                         // default descriptor for accessors:
196                         descriptor = {configurable: true};
197                         accessor = value;
198                     }
199                     //abcdefghijklmnopqrstuvwxyz
200                     for(var i in modifiers) switch(modifiers[i]){
201                         case    'const': descriptor.writable     = false; break;
202                         case    'final': descriptor.configurable = false; descriptor.writable = false; break;
203                         case      'get': descriptor.get          = value; break;
204                         case   'hidden': descriptor.enumerable   = false; break;
205                         case      'set': descriptor.set          = value; break;
206                         case 'writable': descriptor.writable     = true;  break;
207                     }
208                                    
209                     /// define getters/setters:
210                     
211                     if(accessor){
212                         if(typeof accessor == 'string'){
213                             var hiddenPropertyName = accessor;
214                             if(typeof descriptor.get == 'string'){
215                                 descriptor.get = function getter(){
216                                     return this[hiddenPropertyName];
217                                 }
218                             }else{
219                                 descriptor.set = function setter(newValue){
220                                     this[hiddenPropertyName] = newValue;
221                                 }
222                             }
223                         }
224                         // A property cannot both have accessors and be writable or have a value:
225                         //delete descriptor.value;
226                         //delete descriptor.writable;
227                         
228                         if(descriptor.get) value = undefined;// do not allow to hide getter (as default for method)
229                     }     
230                     
231                     /// `once` modifier:
232                     
233                     if( modifiers.indexOf('once') >= 0 ){
234                         var accessorName = descriptor.set ? 'set' : descriptor.get ? 'get' : '';
235                         if( accessorName ){
236                             void function(name, accessor){
237                                 descriptor[accessorName] = function once(v){
238                                     var newValue = accessor.call(this, v);
239                                     var descriptor = Object.create($defaultDescriptor, {value: {value: newValue }});
240                                     Object.defineProperty(this, name, descriptor);
241                                     return newValue;
242                                 }
243                             }(name, accessor);
244                         }else{
245                             console.warn('$object.define: `once` modifier can be used only with `get` or `set` modifier.')
246                         }
247                     }
248                 }
249             }
250             
251             if(! accessor) descriptor.value = value;
252 
253             /// hide methods and private properties:
254             
255             if(hidingAllowed && typeof(value)=='function' || name[0]=='_'){
256                 descriptor.enumerable = false;
257             }
258 
259             /// constants:
260             
261             if(name.toUpperCase() == name){
262                 descriptor.writable = false;
263             }
264             
265             ///
266             
267             descriptors[name] = descriptor;
268         }
269         
270         /// constructor:
271         
272         if( descriptors.hasOwnProperty('constructor') ){
273             var $prototype = this;
274             var constructor = descriptors.constructor.value;
275             if(typeof constructor == 'string'){
276                 var typeName = constructor;
277                 // isn't slow, see http://jsperf.com/eval-vs-new-function-vs-function
278                 constructor = Function(
279                     'return function '+typeName+'(){return this.applySuper(arguments)}'
280                 )();
281                 constructor.typeName = typeName;
282                 descriptors.constructor.value = constructor;
283 
284             }else if(!constructor.typeName){
285                 constructor.typeName = constructor.name || 'CloneOf'+$prototype.constructor.typeName;//+n
286             }
287 
288             constructor.prototype = $prototype;
289             descriptors.constructor.enumerable = false;
290         }
291 
292         ///
293         
294         return descriptors;
295     },

299      * @returns {*}

302      * @this {Object} Instance only.
303      * @protected

306     applySuper: function(/** Array|string='constructor' */ methodName, /** Array= */args){
307         if(typeof(methodName) != 'string'){
308             args = arguments[0];
309             methodName = 'constructor';
310         }//</arguments>
311 
312         '__super__' in this || this.getSuper(true);

315         var savedSuper = this.__super__;
316         // set super to next by prototype chain, in case if method also call applySuper
317         this.__super__ = Object.getPrototypeOf(savedSuper);
318         // apply method
319         var returned   = savedSuper[methodName].apply(this, args);
320         // restore super
321         this.__super__ = savedSuper;
322 
323         return returned;
324     },

416     /**
417      * Return new object, that have a copies of all own properties of this object.  
418      * Arguments can be passed in any order (recognized by type).
419      * @example
420      * var $collection = $object.clone(),
421      *     $users = $collection.clone();
422      *
423      * // $users full prototype chain:
424      * $users -> $collection -> $object -> Object.prototype -> null
425      *
426      * // see prototype chains produced by copy:
427      *
428      * $users.copy():
429      *     ~$users -> $object
430      *
431      * $users.copy(rootPrototype:Array):
432      *     ~$users -> Array.prototype
433      *
434      * $users.copy(rootPrototype:$parent, parentsLevel:Infinity):
435      *     ~$users -> ~$collection -> ~$object -> $parent
436      *
437      * $users.copy(rootPrototype:$parent, parentsLevel:Infinity, mixParents:true):
438      *     ~($users + $collection + $object) -> $parent
439      *
440      * // where ~$users:
441      * New plain object, that have a copy of every own $user property.
442      *
443      * @this {Object} Instance or prototype.
444      * 
445      * @param deepMethod
446      *        How to process inner objects. Can be:        
447      *        "deepCopy"  - see `{@link #deepCopy}`   
448      *        "deepClone" - see `{@link #deepClone}`  
449      *        "" - do nothing (default).
450      *
451      * @param rootPrototype
452      *        The root prototype for created object prototype chain. If it is Constructor, Constructor.prototype will be used instead.  
453      *        By default - $object;
454      *
455      * @param parentsLevel
456      *        How many parents should be included. By default - zero.  
457      *        Set this to Infinity if you want to copy all object parents properties up to $object.
458      *
459      * @param mixParents
460      *        Should be false if objects have methods, that call `{@link #applySuper}`.
461      *        If true, all own properties of all objects will be directly attached to the one returned object.  
462      *        False by default.
463      *
464      * @returns {Object}
465      * @memberOf $object#
466      */
467     copy: function(
468         /** ('deepClone'|'deepCopy')=""      */deepMethod,
469         /**     (Object|Constructor)=$object */rootPrototype,
470         /**                   number=0       */parentsLevel,
471         /**                  boolean=false   */mixParents
472         //**                Array=all */propertiesList
473     ){
474         for(var i=0, value=arguments[i], length=arguments.length; i<length; value=arguments[++i]) switch(typeof value){
475             case 'string':   deepMethod    = value; break;
476             case 'function': rootPrototype = value.prototype; break;
477             case 'object':   rootPrototype = value; break;
478             case 'number':   parentsLevel  = value; break;
479             case 'boolean':  mixParents    = value; break;
480         }
481 
482         if( deepMethod != 'deepCopy' || deepMethod != 'deepClone'){
483             deepMethod = null;
484         }
485 
486         if(typeof rootPrototype == 'undefined'){
487             rootPrototype = $object;
488         }
489         //</arguments>
490 
491         var sourceObj = this;
492         if( typeof(sourceObj)=='function' && Object.getOwnPropertyNames(sourceObj.prototype) ){
493             sourceObj = sourceObj.prototype;
494         }
495         var newObj    = Object.create(rootPrototype);
496         var updateObj = newObj;
497 
498         var sourceObjects = [];
499         do{
500             sourceObjects.push(sourceObj);
501             sourceObj = Object.getPrototypeOf(sourceObj);
502         }while(parentsLevel-- && sourceObj != Object.prototype);
503 
504         sourceObjects = sourceObjects.reverse();
505 
506         for(var i=0, length=sourceObjects.length; i<length; i++){
507             sourceObj = sourceObjects[i];
508             var ownPropertyNames = Object.getOwnPropertyNames(sourceObj);
509             if (!mixParents && i && ownPropertyNames.length){
510                 updateObj = $object.apply(updateObj, 'clone');
511             }
512 
513             // copy all own properties:
514             for(var j=0, jLength=ownPropertyNames.length; j<jLength; j++){
515                 var name = ownPropertyNames[j];
516                 var descriptor = Object.getOwnPropertyDescriptor(sourceObj, name);
517 
518                 if( deepMethod && typeof sourceObj[name] == 'object' && sourceObj[name] !== null){
519                     descriptor.value = $object[deepMethod].call(sourceObj[name]);
520                 }
521 
522                 Object.defineProperty(updateObj, name, descriptor);
523             }
524         }
525 
526         return updateObj;
527     },

596      * @this {Object} Instance or prototype.
597      * @param obj

604     concat: function(/** Object */obj, /** (Array|boolean)= */allProperties){
605         if(allProperties == undefined) allProperties = obj.getOwnPropertyNames();
606         else if(typeof allProperties != 'object' && allProperties == true){
607             allProperties = obj.getOwnPropertyNames();
608             var proto = obj;
609             while( (proto = Object.getPrototypeOf(proto)) !== Object.prototype ){
610                 allProperties = allProperties.concat( proto.getOwnPropertyNames() );
611             }
612         }
613 
614         proto = obj;
615         do for(var i=0, length=allProperties.length; i<length; i++){
616             var name = allProperties[i];
617             var descriptor = Object.getOwnPropertyDescriptor(proto, name);
618             if( descriptor ){
619                 Object.defineProperty(this, name, descriptor);
620                 delete allProperties[i];
621             }
622         }
623         while(allProperties.length && (proto = Object.getPrototypeOf(proto)) !== Object.prototype);
624 
625         return this;

638     apply: function(/** Object */withObj, /** string */methodName, /** Array= */args, /** Object= */asObj){
639         if(!asObj){
640             asObj = typeof(withObj[methodName])=='function' 
641                  && withObj[methodName].length == this[methodName].length 
642                  && withObj

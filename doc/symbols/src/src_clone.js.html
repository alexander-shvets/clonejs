1 /**
  2  * @title   clone.js - The true prototype-based OOP framework.
  3  * @version 0.6.0 alpha
  4  * @author  Alex Shvets
  5  * @see     https://github.com/quadroid/clonejs
  6  *
  7  * @class
  8  *     This is the framework that implements the true prototype-based OOP paradigm in JS.<br>
  9  *     It's based on the new ECMA Script 5 features like Object.create and property descriptors.
 10  *
 11  *     <p><a href="https://github.com/quadroid/clonejs">Project on GitHub</a>
 12  *
 13  *     <p><b>Naming conventions</b>
 14  *     <p>Var names, prefixed by "<b>$</b>", contain object, used as prototype for other objects. For example:<br>
 15  *        var $array = Array.prototype, $myType = {}, <br>
 16  *            myTypeInstance = Clone($myType);
 17  *     <p>Properties, prefixed by "<b>_</b>", are private.
 18  *     <p>
 19  * @description
 20  *     With this library you should forget about classes. Use prototypes instead.
 21  * @example
 22  *     var $myType = $object.clone({
 23  *         '(final)  notConfigurableAndNotWritable': true,
 24  *         '(writable final)   notConfigurableOnly': null,
 25  *         '(hidden final get) notEnumerableGetter': function(){},
 26  *         '(hidden)                 notEnumerable': true,
 27  *         '(const)                       constant': 'not writable',
 28  *                                  publicProperty : 1,
 29  *                                           _item : null,// private property (not enumerable)
 30  *                                     '(get) item': function() { return this._item },
 31  *                                     '(set) item': function(v){ this._item = v },
 32  *                      '(get) publicPropertyAlias': 'publicProperty',// automatically create getter for publicProperty
 33  *                                     constructor : function MyType(){
 34  *                                                       var obj = this.applySuper(arguments);
 35  *                                                       // do something with obj...
 36  *                                                       return obj;
 37  *                                                   }
 38  *     });
 39  *     var myTypeInstance = $myType.create({publicProperty: 2});
 40  *     assert( $myType.isPrototypeOf(myTypeInstance) );
 41  *     assert( $myType.publicPropertyAlias === $myType.publicProperty );
 42  *
 43  *     var $myArray1 = $object.clone.call(Array.prototype, {customMethod: function(){}});
 44  *     var $myArray2 = Clone.makeFom(Array, {customMethod: function(){}});
 45  *
 46  *     var myObj = {a:1, b:2, c:3};
 47  *     var cloneOfMyObj = $object.clone.call(myObj);
 48  *     cloneOfMyObj.a = 11; // myObj.a still == 1

 51     /**
 52      * Create a clone of object. See <a href="http://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/create">Object.create</a> for details.
 53      * @see <a href="http://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/create">Object.create</a>
 54      * @returns {$object}
 55      * @memberof $object#
 56      */
 57     clone: function(/** Object= */properties, /** PropertyDescriptor= */defaultDescriptor){
 58         if(arguments.length){
 59             var descriptors = $object.describe.apply(null, arguments);
 60 
 61             if( descriptors.hasOwnProperty('constructor') ){
 62                 var constructor = descriptors.constructor.value;
 63                 if(typeof constructor == 'string'){
 64                     descriptors.constructor.value = function CustomType(){
 65                         return this.applySuper(arguments);
 66                     }
 67                 }
 68                 descriptors.constructor.value.prototype = this;
 69             }
 70         }
 71         return Object.create(this, descriptors);
 72     },

111     /**
112      * Translate object to property descriptors.
113      * <p>For example, {a:{}, b:2} will be translated to something like {a: {value: {}}, b: {value: 2}}.
114      * <p>Functions (except getters) and properties prefixed by "_" will be automatically marked as non-enumerable.
115      * <p>You can prefix your property names by (get|set|const|final|hidden|writable).
116      *     <li>(get) - define getter, if string passed, the getter will be auto generated.
117      *     <li>(set) - define setter, if string passed, the setter will be auto generated.
118      *     <li>(const) - make property unwritable.
119      *     <li>(final) - make property unwritable, and prevent it deleting and descriptor modifications.
120      *     <li>(hidden) - make property non-enumerable.
121      *     <li>(writable) - make property writable (use with final).
122      *
123      * @param {Object} properties
124      * @param {PropertyDescriptor=} defaultDescriptor The default property descriptor.
125      * @returns {{PropertyDescriptor}} Property descriptors.
126      *
127      * @static
128      */
129     describe: function(properties, defaultDescriptor){
130         var descriptors = {};
131 
132         var $defaultDescriptor = defaultDescriptor ? defaultDescriptor : {
133             configurable: true,
134             enumerable: true,
135             writable: true
136         };
137 
138         var hidingAllowed = !(defaultDescriptor && defaultDescriptor.enumerable);
139 
140         for(var name in properties){
141             var value = properties[name];
142             var descriptor = Object.create($defaultDescriptor);
143 
144             if( name[0]=='(' ){
145                 //TODO: fix regexp to not mach the '(getset) property'
146                 var matches = name.match(/^\((((get|set|const|hidden|final|writable) *)+)\) +(.+)$/);
147                 if( matches ){
148                     var prefixes = matches[1].split(' ').sort();
149                     name = matches[4];
150 
151                     if(descriptors[name]) descriptor = descriptors[name];
152 
153                     for(var i in prefixes) switch(prefixes[i]){
154                         case    'const': descriptor.writable     = false; break;
155                         case    'final': descriptor.configurable = false; descriptor.writable = false; break;
156                         case      'get': descriptor.get          = value; break;
157                         case   'hidden': descriptor.enumerable   = false; break;
158                         case      'set': descriptor.set          = value; break;
159                         case 'writable': descriptor.writable     = true;  break;
160                     }
161                 }
162             }
163 
164             if(descriptor.get || descriptor.set){
165                 if(typeof value == 'string'){
166                     var hiddenPropertyName = value;
167                     if(typeof descriptor.get == 'string'){
168                         descriptor.get = function getter(){
169                             return this[hiddenPropertyName];
170                         }
171                     }else{
172                         descriptor.set = function setter(newValue){
173                             this[hiddenPropertyName] = newValue;
174                         }
175                     }
176                 }
177                 descriptor.value = undefined;
178                 if(descriptor.get) value = undefined;// do not allow to hide getter by default
179             }else{
180                 descriptor.value = value;
181             }
182 
183             // hide methods and private properties:
184             if(hidingAllowed && typeof(value)=='function' || name[0]=='_'){
185                 descriptor.enumerable = false;
186             }
187     //        // constants:
188     //        if(name.toUpperCase() == name){
189     //            descriptor.writable = false;
190     //        }
191             descriptors[name] = descriptor;
192         }
193     //    if( descriptors.hasOwnProperty('constructor')){
194     //        descriptors.constructor.enumerable = false;
195     //    }
196         return descriptors;
197     },

199     /**
200      * Apply method of super object (prototype) to this object.
201      * @returns {*}
202      * @see $object#__super__
203      * @see $object#callSuper
204      * @memberof $object#
205      */
206     applySuper: function(/** string='constructor'|Array */ methodName, /** Array= */args){
207         if(typeof(methodName) != 'string'){
208             if( arguments[0] instanceof Array){
209                 args = arguments[0];
210             }
211             methodName = 'constructor';
212         }//</arguments>
213 
214         /* if not */('__super__' in this) || this.defineProperty(
215             '__super__', {value: Object.getPrototypeOf(Object.getPrototypeOf(this)), writable:!0,configurable:!0}
216             /**
217              * Link to the object prototype.
218              * Dynamically changed to next by prototype chain, while applySuper method executing.
219              * System property. <b>Use it only for debug purposes</b>.
220              * @name  __super__
221              * @type  {?Object}
222              * @see $object#applySuper
223              * @memberOf $object#
224              */
225         );
226 
227         // save super
228         var savedSuper = this.__super__;
229         // set super to next by prototype chain, in case if method also call applySuper
230         this.__super__ = Object.getPrototypeOf(savedSuper);
231         // apply method
232         var returned   = savedSuper[methodName].apply(this, args);
233         // restore super
234         this.__super__ = savedSuper;
235 
236         return returned;
237     },

275     createSuperSafeCallback: function(/** Function|string */functionOrMethodName, /** Object= */boundThis){
276         if(typeof functionOrMethodName == 'string'){
277             var fn = this[functionOrMethodName];
278             if(typeof boundThis == 'undefined') boundThis = this;
279         }else{
280             fn = functionOrMethodName;
281         }
282         //</arguments>
283 
284         var self = this;
285         var callbackSuper = this.__super__;
286 
287         return function superSafeCallback(){
288 
289             if(self.__super__ === callbackSuper){
290                 return fn.apply(boundThis||this, arguments);
291 
292             }else{
293                 var savedSuper = self.__super__;
294                 self.__super__ = callbackSuper;

322      * Mix two or more objects.
323      *
324      * @param objectToMix

337     mix: function(
338         /** Object|FunctionType|Array.<(Object|FunctionType)> */
339         objectToMix,
340         /**            number=0 */parentsLevel,
341         /**        boolean=true */copyNesting
342     ){
343 //        if( typeof(this)=='function' && Object.getOwnPropertyNames(this.prototype) ){

354             mixedObjects = objectToMix;
355         }else{
356             // get parents:

362 
363         mixedObjects = mixedObjects.reverse();

365         var updateObj = this;
366 
367         for(var objIdx in mixedObjects){
368             obj = mixedObjects[objIdx];
369 
370             if( typeof(obj)=='function' && Object.getOwnPropertyNames(obj.prototype) ){

416      * @example
417      *     var myObj1 = $object.clone({join: function(sep){} });
418      *     var myObj2 = $object.clone({join: function(){}    });
419      *     assert( myObj1.can('split').like( Array.prototype ) === true  );
420      *     assert( myObj2.can('split').like( Array.prototype ) === false );
421      *     assert( myObj1.can('split').as(   Array.prototype ) === false );
422      *     assert( $object.can.call(new Array, 'split').as(new Array) === true  );
423      *
424      * @returns {{like: function(Object):boolean, as: function(Object):boolean}}

426      */
427     can: function(/** string */method, /** number=false */not){
428         var obj1 = this;
429         return {
430             /** @ignore *///jsdoc3
431             valueOf: function(){
432                 // a ^ b == a ? !b : b
433                 return not ^ typeof( obj1[method] )=='function';
434             },
435 
436             like: function(obj2){
437                 var obj2Method = obj2[method];
438                 return this.valueOf()
439                     && not ^ (
440                     typeof(obj2Method)=='function' &&
441                         obj2Method.length == obj1[method].length
442                     );
443             },
444 
445             as: function(obj2){
446                 return this.valueOf()
447                     && not ^ obj1[method] === obj2[method];
448             }
449         }

1 'use strict';
  2 /**
  3  * @title   clone.js - the true prototype-based JavaScript micro-framework.
  4  * @version 0.7.0 beta
  5  * @author  Alex Shvets
  6  *
  7  * @class
  8  * This is the framework that implements the true [prototype-based OOP⠙][1] paradigm in JS. 
  9  * It's based on the ECMA Script 5 features like [Object.create⠙][2] and [property descriptors⠙][3].
 10  *    [1]: http://en.wikipedia.org/wiki/Prototype-based_programming
 11  *    [2]: https://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/create
 12  *    [3]: https://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/defineProperty
 13  *
 14  * <p>[View on GitHub](http://github.com/quadroid/clonejs#readme)
 15  *
 16  * ### Naming conventions
 17  *  
 18  * Var names, **prefixed by "$"**, contain object, used as prototype for other objects.  
 19  * For example:
 20  * <code>
 21  * var $array = Array.prototype, $myType = {},
 22  *     myTypeInstance = Object.create($myType);// $object.apply($myType, 'create');
 23  * </code>
 24  *
 25  * Properties, **prefixed by "_"**, are private.
 26  * 
 27  * [![githalytics.com alpha](https://cruel-carlota.pagodabox.com/3110be9614da5cb337ebd483c187010f "githalytics.com")](http://githalytics.com/quadroid/clonejs)
 28  *
 29  * @description
 30  * The main difference with other class-producing tools like `Ext.define`, `dojo.declare`, `Backbone.Model.extend`
 31  * is that `$object.clone` will return an object (prototype with defined constructor-function) instead of function
 32  * (with defined prototype-object). So, you don't need for instantiation, you can just start using the cloned object right now.
 33  * But, if you need more than one instance, you can create it by `$yourProto.create`.
 34  *
 35  * @example
 36  *
 37  * var myObj = {a:1, b:2, c:3};
 38  * var cloneOfMyObj = $object.apply(myObj, 'clone');
 39  * cloneOfMyObj.a = 11; // myObj.a still == 1
 40  * myObj.b = 22; // cloneOfMyObj.b will be also changed to 22
 41  *
 42  * var $myType = $object.clone({
 43  *        '(final)          property1': "not configurable and not writable",
 44  *        '(writable final) property2': "not configurable only",
 45  *        '(hidden)         property3': "not enumerable",
 46  *        '(const)           constant': "not writable",
 47  *                          property4 : "simple property",
 48  *        '(get)       property3alias': 'property3',// automatically create getter
 49  *                              _item : "private property (not enumerable)",
 50  *                       '(get)  item': function() { return this._item },
 51  *                       '(set)  item': function(v){ this._item = v    },
 52  *                        constructor : function MyType(){
 53  *                                          this.applySuper(arguments);
 54  *                                          // do something...
 55  *                                      }
 56  *    });
 57  * var myTypeInstance = $myType.create({property4: "initialize simple property"});
 58  * assert( $myType.isPrototypeOf(myTypeInstance) );
 59  * assert( $myType.property3alias === $myType.property3 );
 60  *
 61  * var $myArray1 = $object.clone.call(Array.prototype, {customMethod: function(){}});
 62  * var $myArray2 = $object.copy(Array).setProperties({customMethod: function(){}});
 63  *
 64  */

 67     /**
 68      * Create a clone of object.
 69      * @see <a href="http://developer.mozilla.org/en-US/docs/JavaScript/Reference/Global_Objects/Object/create">Object.create⠙</a>
 70      * @returns {$object}
 71      * @memberOf $object#
 72      */
 73     clone: function(/** Object= */properties, /** PropertyDescriptor= */defaultDescriptor){
 74         if(arguments.length){
 75             var descriptors = $object.describe.apply(null, arguments);
 76 
 77             if( descriptors.hasOwnProperty('constructor') ){
 78                 var constructor = descriptors.constructor.value;
 79                 if(typeof constructor == 'string'){
 80                     var typeName = constructor;
 81                     constructor = descriptors.constructor.value = function CustomType(){
 82                         return this.applySuper(arguments);
 83                     }
 84                     constructor.typeName = typeName;
 85                     
 86                 }else if(!constructor.typeName){
 87                     constructor.typeName = constructor.name || this.constructor.typeName + '_clone';//+n
 88                 }
 89 
 90                 constructor.prototype = this;
 91             }
 92         }
 93         return Object.create(this, descriptors);
 94     },

136     /**
137      * Translate object to property descriptors.
138      *
139      * For example, `{a:{}, b:2}` will be translated to something like `{a: {value: {}}, b: {value: 2}}`.
140      * Functions (except getters) and properties prefixed by "_" will be automatically marked as non-enumerable. 
141      *
142      * You can prefix your property names by `(get|set|const|final|hidden|writable)`:
143      *
144      * + `(get)`      - define getter, if string passed, the getter will be auto generated.
145      * + `(set)`      - define setter, if string passed, the setter will be auto generated.
146      * + `(const)`    - make property unwritable.
147      * + `(final)`    - make property unwritable, and prevent it deleting and descriptor modifications.
148      * + `(hidden)`   - make property non-enumerable.
149      * + `(writable)` - make property writable (use with final).
150      *
151      * @param properties
152      * @param defaultDescriptor The default property descriptor.
153      * @returns {{PropertyDescriptor}} Property descriptors.
154      * @static
155      * @memberOf $object
156      */
157     describe: function(/** Object */properties, /** PropertyDescriptor= */defaultDescriptor){
158         var descriptors = {};
159 
160         var $defaultDescriptor = defaultDescriptor ? defaultDescriptor : {
161             configurable: true,
162             enumerable: true,
163             writable: true
164         };
165 
166         var hidingAllowed = !(defaultDescriptor && defaultDescriptor.enumerable);
167 
168         for(var name in properties){
169             var value = properties[name];
170             var descriptor = Object.create($defaultDescriptor);
171 
172             if( name[0]=='(' ){
173                 //TODO: fix regexp to not mach the '(getset) property'
174                 var matches = name.match(/^\((((get|set|const|hidden|final|writable) *)+)\) +(.+)$/);
175                 if( matches ){
176                     var prefixes = matches[1].split(' ').sort();
177                     name = matches[4];
178 
179                     if(descriptors[name]) descriptor = descriptors[name];
180 
181                     for(var i in prefixes) switch(prefixes[i]){
182                         case    'const': descriptor.writable     = false; break;
183                         case    'final': descriptor.configurable = false; descriptor.writable = false; break;
184                         case      'get': descriptor.get          = value; break;
185                         case   'hidden': descriptor.enumerable   = false; break;
186                         case      'set': descriptor.set          = value; break;
187                         case 'writable': descriptor.writable     = true;  break;
188                     }
189                 }
190             }
191 
192             if(descriptor.get || descriptor.set){
193                 if(typeof value == 'string'){
194                     var hiddenPropertyName = value;
195                     if(typeof descriptor.get == 'string'){
196                         descriptor.get = function getter(){
197                             return this[hiddenPropertyName];
198                         }
199                     }else{
200                         descriptor.set = function setter(newValue){
201                             this[hiddenPropertyName] = newValue;
202                         }
203                     }
204                 }
205                 descriptor.value = undefined;
206                 if(descriptor.get) value = undefined;// do not allow to hide getter by default
207             }else{
208                 descriptor.value = value;
209             }
210 
211             // hide methods and private properties:
212             if(hidingAllowed && typeof(value)=='function' || name[0]=='_'){
213                 descriptor.enumerable = false;
214             }
215     //        // constants:
216     //        if(name.toUpperCase() == name){
217     //            descriptor.writable = false;
218     //        }
219             descriptors[name] = descriptor;
220         }
221     //    if( descriptors.hasOwnProperty('constructor')){
222     //        descriptors.constructor.enumerable = false;
223     //    }
224         return descriptors;
225     },

232      * @protected
233      * @memberOf $object#

235     applySuper: function(/** string='constructor'|Array */ methodName, /** Array= */args){
236         if(typeof(methodName) != 'string'){
237             if( arguments[0] instanceof Array){
238                 args = arguments[0];
239             }
240             methodName = 'constructor';
241         }//</arguments>
242 
243         /* if not */('__super__' in this) || this.defineProperty(
244             '__super__', {value: Object.getPrototypeOf(Object.getPrototypeOf(this)), writable:!0,configurable:!0}

246              * Link to the instance prototype.
247              * Dynamically changed to next by prototype chain, while `{@link #applySuper}` method executing.
248              * System property. **Use it only for debug purposes**.

250              * @type  {?Object}
251              * @see $object#applySuper
252              * @private
253              * @memberOf $object#
254              */
255         );
256 
257         // save super
258         var savedSuper = this.__super__;
259         // set super to next by prototype chain, in case if method also call applySuper
260         this.__super__ = Object.getPrototypeOf(savedSuper);
261         // apply method
262         var returned   = savedSuper[methodName].apply(this, args);
263         // restore super
264         this.__super__ = savedSuper;
265 
266         return returned;

351     /**
352      * Return new object, that have a copies of all own properties of this object.  
353      * Arguments can be passed in any order (recognized by type).
354      * @example
355      * var $collection = $object.clone(),
356      *     $users = $collection.clone();
357      *
358      * // $users full prototype chain:
359      * $users -> $collection -> $object -> Object.prototype -> null
360      *
361      * // see prototype chains produced by copy:
362      *
363      * $users.copy():
364      *     ~$users -> $object
365      *
366      * $users.copy(rootPrototype:Array):
367      *     ~$users -> Array.prototype
368      *
369      * $users.copy(rootPrototype:$parent, parentsLevel:Infinity):
370      *     ~$users -> ~$collection -> ~$object -> $parent
371      *
372      * $users.copy(rootPrototype:$parent, parentsLevel:Infinity, mixParents:true):
373      *     ~($users + $collection + $object) -> $parent
374      *
375      * // where ~$users:
376      * New plain object, that have a copy of every own $user property.
377      *
378      * @param deepMethod
379      *        How to process inner objects. Can be:        
380      *        "deepCopy"  - see `{@link #deepCopy}`   
381      *        "deepClone" - see `{@link #deepClone}`  
382      *        "" - do nothing (default).
383      *
384      * @param rootPrototype
385      *        The root prototype for created object prototype chain. If it is Constructor, Constructor.prototype will be used instead.  
386      *        By default - $object;
387      *
388      * @param parentsLevel
389      *        How many parents should be included. By default - zero.  
390      *        Set this to Infinity if you want to copy all object parents properties up to $object.
391      *
392      * @param mixParents
393      *        Should be false if objects have methods, that call `{@link #applySuper}`.
394      *        If true, all own properties of all objects will be directly attached to the one returned object.  
395      *        False by default.
396      *
397      * @returns {Object}
398      * @memberOf $object#
399      */
400     copy: function(
401         /** ('deepClone'|'deepCopy')=""      */deepMethod,
402         /**     (Object|Constructor)=$object */rootPrototype,
403         /**                   number=0       */parentsLevel,
404         /**                  boolean=false   */mixParents
405         //**                Array=all */propertiesList
406     ){
407         for(var i=0, value=arguments[i]; i < arguments.length; value=arguments[++i]) switch(typeof value){
408             case 'string':   deepMethod    = value; break;
409             case 'function': rootPrototype = value.prototype; break;
410             case 'object':   rootPrototype = value; break;
411             case 'number':   parentsLevel  = value; break;
412             case 'boolean':  mixParents    = value; break;
413         }
414 
415         if( deepMethod != 'deepCopy' || deepMethod != 'deepClone'){
416             deepMethod = null;
417         }
418 
419         if(typeof rootPrototype == 'undefined'){
420             rootPrototype = $object;
421         }
422 
423         var sourceObj = this;
424         if( typeof(sourceObj)=='function' && Object.getOwnPropertyNames(sourceObj.prototype) ){
425             sourceObj = sourceObj.prototype;
426         }
427         var newObj    = Object.create(rootPrototype);
428         var updateObj = newObj;
429 
430         var sourceObjects = [];
431         do{
432             sourceObjects.push(sourceObj);
433             sourceObj = Object.getPrototypeOf(sourceObj);
434         }while(parentsLevel-- && sourceObj != Object.prototype);
435 
436         sourceObjects = sourceObjects.reverse();
437 
438         for(var i=0; i < sourceObjects.length; i++){
439             sourceObj = sourceObjects[i];
440             var ownPropertyNames = Object.getOwnPropertyNames(sourceObj);
441             if (!mixParents && i && ownPropertyNames.length){
442                 updateObj = $object.apply(updateObj, 'clone');
443             }
444 
445             // copy all own properties:
446             for(var p=0; p < ownPropertyNames.length; p++){
447                 var name = ownPropertyNames[p];
448                 var descriptor = Object.getOwnPropertyDescriptor(sourceObj, name);
449 
450                 if( deepMethod && typeof sourceObj[name] == 'object' && sourceObj[name] !== null){
451                     descriptor.value = $object[deepMethod].call(sourceObj[name]);
452                 }
453 
454                 Object.defineProperty(updateObj, name, descriptor);
455             }
456         }
457 
458         return updateObj;
459     },

772 /**
773  * @namespace
774  *    The root namespace.
775  * @description
776  * If you run the following code:
777  * <code>
778  * ns.extend('collection', {name: 'collection'});
779  * ns.collection.extend('arrayCollection', {name: 'arrayCollection'});
780  * </code>
781  * The structure of the namespace will be:
782  * <code>
783  * ns == {
784  *     $collection: {name: 'collection'},
785  *     collection: {
786  *         $arrayCollection: {name: 'arrayCollection'},
787  *         arrayCollection: {
788  *             prototype: {name: 'arrayCollection'},
789  *             extend: ns.extend
790  *         },
791  *         prototype: {name: 'collection'},
792  *         extend: ns.extend
793  *     },
794  *     prototype: $object,
795  *     extend: ns.extend,
796  *     put: ns.put
797  * }
798  * </code>
799  */
800 var ns = {
801     /** 
802      * The prototype of every object in this namespace (default - `{@link $object}`).
803      * @name prototype
804      * @memberOf ns# */
805     prototype: $object,
806     
807     /**
808      * Create prototype as a part of namespace.
809      * @see $object#clone
810      * @memberOf ns# */
811     extend: function extend(/** string */nsName, /** Object */prototype, /** PropertyDescriptor= */defaultDescriptor){
812         if( nsName.indexOf('.') > 0 ){
813             var currentNS = this;
814             var nameParts = nsName.split('.');
815             nsName = nameParts.pop();
816             nameParts.forEach(function(namePart){
817                 if(!(namePart in currentNS)){
818                     currentNS.extend(namePart);
819                 }
820                 currentNS = currentNS[namePart];
821             });
822         }
823 
824         var typeName  = nsName[0].toUpperCase() + nsName.substr(1);
825         if(!prototype.hasOwnProperty('constructor')){
826             prototype.constructor = typeName;
827         }
828         var $newProto = $object.isPrototypeOf(prototype) ? prototype : this.prototype.clone(prototype, defaultDescriptor);
829         $newProto.constructor.typeName = typeName;
830     
831         var newNS = {
832             prototype: $newProto,
833             extend: extend
834         };
835         this['$'+nsName] =  $newProto;
836         this[nsName] = newNS;
837         
838         return newNS;
839     },
840 
841     /**
842      * Put object (and all its parents) into this namespace.
843      * Name will be get from prototype.constructor.name.
844      * @returns {ns}
845      * @static */
846     put: function(/** Object */prototype){
847         var eachNS = this;
848 
849         var prototypes = prototype.getPrototypes();
850         prototypes.push(prototype);
851         prototypes.forEach(function(proto){
852             var name = proto.constructor.typeName || proto.constructor.name;
853             name = name[0].toLowerCase() + name.substr(1);
854             if(!(name in eachNS)){
855                 eachNS.extend(name, proto);
856             }
857             eachNS = eachNS[name];    
858         });
859         
860         return eachNS;
861     }

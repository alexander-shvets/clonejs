  2 /**
  3  * @title   clone.js - The true prototype-based OOP framework.
  4  * @version 0.4.5
  5  * @author  Alex Shvets
  6  * @see     https://github.com/quadroid/clonejs
  7  *
  8  * @class
  9  *     This is the framework that implements the true prototype-based OOP paradigm in JS.
 10  *     It based on the new ECMA Script 5 features like Object.create and property descriptors.
 11  *
 12  *     <p><b>Naming conventions</b>
 13  *     <p><b>Var names, prefixed by "$"</b>, contain object, used as prototype for other objects. For example:<br>
 14  *        var $array = Array.prototype, $myType = {}, <br>
 15  *            myTypeInstance = Clone($myType);
 16  *     <p>
 17  *     <p><b>properties, prefixed by "_"</b>, are private.
 18  *     <p>
 19  * @description
 20  *     If called as function, return a clone of <i>baseObj</i>, and set it's own properties.
 21  *     If called by <i>new</i> operator, constructor will take 2 arguments (properties and defaultDescriptor).
 22  *
 23  * @example
 24  *     var $myType = Clone.make({
 25  *         constructor: Clone.defineType('MyType', function(){
 26  *             this.applySuper(arguments);
 27  *             // do something with this...
 28  *         }),
 29  *         property: 1
 30  *     });
 31  *     var myTypeInstance = $myType.create();
 32  *     assert( $myType.isPrototypeOf(myTypeInstance) );
 33  *
 34  *     var $myArray1 = Clone(Array.prototype, {customMethod: function(){}});
 35  *     var $myArray2 = Clone.makeFomClass(Array, {customMethod: function(){}});
 36  *
 37  *     var myObj = {a:1, b:2, c:3};
 38  *     var cloneOfMyObj = Clone(myObj);
 39  *     cloneOfMyObj.a = 11; // myObj.a still == 1
 40  *
 41  *     var myClone = new Clone();
 42  *     myClone.defineProperties({a:1, b:2, c:3});
 43  *
 44  *     @param {Object} baseObj
 45  *            Does not used, if called by new operator.
 46  *
 47  *     @param {Object=} properties
 48  *            Own properties of the created object.
 49  *
 50  *     @param {PropertyDescriptor=} defaultDescriptor
 51  *            Default property descriptor for properties.
 52  *
 53  * @return {Object} The new clone.
 54  */
 55 var Clone = function Clone(baseObj, /** Object= */properties, /** PropertyDescriptor= */defaultDescriptor){
 56 
 57     if(this && this !== (window || global) ){
 58     // Called as constructor (by new operator or used call/apply):
 59         // We assume that the new object is already created,
 60         // and we need only to set its own properties (if specified).
 61         if(arguments.length){
 62             Array.prototype.unshift.call(arguments, this);
 63             Function.call.apply(Clone.prototype.defineProperties, arguments);
 64         }
 65         return this;// need for call/apply
 66 
 67     }else{
 68     // Called like a regular function:
 69         if(arguments.length == 0) baseObj = Clone.prototype;
 70 
 71         return Clone.prototype.clone.call(baseObj, properties, defaultDescriptor);
 72     }
 73 };

 75 /**
 76  * Translate object to property descriptors.
 77  * <p>For example, {a:{}, b:2} will be translated to something like {a: {value: {}}, b: {value: 2}}.
 78  * <p>Functions, properties prefixed by "_", and constructor will be automatically marked as non-enumerable.
 79  *
 80  * @param {Object} properties
 81  * @param {PropertyDescriptor=} defaultDescriptor The default property descriptor for properties.
 82  * @returns {{PropertyDescriptor}} Properties descriptors.
 83  *
 84  * @static
 85  */
 86 Clone.describe = function(properties, defaultDescriptor){
 87     var descriptors = {};
 88 
 89     if(!defaultDescriptor){
 90         /*
 91          * @typedef {Object}
 92          * @name PropertyDescriptor */
 93         defaultDescriptor = {
 94             value: undefined,
 95             configurable: true,
 96             writable: true,
 97             enumerable: true
 98         };
 99     }
100 
101     for(var property in properties){
102         var value = properties[property];
103         var descriptor = descriptors[property] = Object.create(defaultDescriptor);
104         descriptor.value = value;
105 
106         // methods and private properties should be non-enumerable:
107         if(typeof(value)=='function' || property[0]=='_'){
108             descriptor.enumerable = false;
109         }
110     }
111     if( descriptors.hasOwnProperty('constructor')){
112         descriptors.constructor.enumerable = false;
113     }
114     return descriptors;
115 };

205 /**
206  * Use this to check if method of one object is the same as in the another object.
207  * @example
208  *     var myObj = {split: function(){} };
209  *     Clone.can(myObj, 'split').like(new Array);
210  *
211  * @returns {{valueOf: Function, like: function(Object), as: function(Object)}}
212  */
213 Clone.can = function(/** Object */obj1, /** string */method, /** number=false */not){
214     return {
215         /** @ignore *///jsdoc3
216         valueOf: function(){
217             // a ^ b == a ? !b : b
218             return not ^ typeof( obj1[method] )=='function';
219         },
220 
221         like: function(obj2){
222             var obj2Method = obj2[method];
223             return this.valueOf()
224                 && not ^ (
225                     typeof(obj2Method)=='function' &&
226                     obj2Method.length == obj1[method].length
227                 );
228         },
229 
230         as: function(obj2){
231             return this.valueOf()
232                 && not ^ obj1[method] === obj2[method];
233         }
234     }
235 };
236 /** @see Clone.can */
237 Clone.cant = function(/** Object */obj1, /** string */method){

241 /**
242  * Mix two or more objects.
243  *
244  * @param root
245  *        The root object. Will be modified, if copyNesting does not true.
246  *
247  * @param objectToMix
248  *        Object(s) to mix. If it is FunctionType, FunctionType.prototype will be used instead.
249  *
250  * @param parentsLevel
251  *        Set this to Infinity if you want to copy all objectToMix parents properties up to Object.prototype.
252  *
253  * @param copyNesting
254  *        Should be true if objectsToMix have methods, that call {@link Clone#applySuper}
255  *        If not true, all own properties of all objects will be directly attached to the one root object.
256  *
257  * @returns {Object} Modified root object if copyNesting, else - the new object based on objectsToMix copies and root.
258  */
259 Clone.mix = function(
260     /**       Object|FunctionType */root,
261     /** Object|FunctionType|Array */objectToMix,
262     /**                  number=0 */parentsLevel,
263     /**              boolean=true */copyNesting
264 ){
265     if( typeof(root)=='function' && Object.getOwnPropertyNames(root.prototype) ){
266         root = root.prototype;
267     }
268     if( copyNesting === undefined ){
269         if(typeof parentsLevel == 'boolean'){
270             copyNesting   = parentsLevel;
271             parentsLevel  = 0;
272         }else copyNesting = true;
273     }
274 
275     if(objectToMix instanceof Array){
276         mixedObjects = objectToMix;
277         //objectToMix = mixedObjects.shift();
278     }else{
279         // get parents:
280         var obj = objectToMix;
281         var mixedObjects = [];
282         do  mixedObjects.push(obj);
283         while( parentsLevel-- && (obj = Object.getPrototypeOf(obj)) != Object.prototype);
284     }
285 
286     mixedObjects = mixedObjects.reverse();
287 
288     var updateObj = root;
289 
290     for(var objIdx in mixedObjects){
291         obj = mixedObjects[objIdx];
292 
293         if( typeof(obj)=='function' && Object.getOwnPropertyNames(obj.prototype) ){
294             obj = obj.prototype;
295         }
296 
297         var ownPropertyNames = Object.getOwnPropertyNames(obj);
298 
299         if(copyNesting && ownPropertyNames.length){
300             updateObj = Clone(updateObj);
301         }
302 
303         // copy all own properties from obj to updateObj:
304         for(var i=0; i < ownPropertyNames.length; i++){
305             var name = ownPropertyNames[i];
306             var descriptor = Object.getOwnPropertyDescriptor(obj, name);
307             Object.defineProperty(updateObj, name, descriptor);
308         }
309     }
310 
311     return updateObj;
312 };
